通过mousedown时记录下鼠标的位置x1，y1，mouseup时记录下鼠标的位置x2,y2，
通过计算判断拖动距离来判断此时是触发点击事件还是拖动事件。

var _val=Math.sqrt((x2-x1)(x2-x1)+(y2-y1)(y2-y1));
//判断
if(_val>=0&&_val<=2){
dragFlag=true;
}else{
dragFlag=false;
}
